Section CommonRules.

  Context `{SL : STRLANG}.

  (* Reflexivity axioms *)
  Definition atrefl : rule := ([],
                               FS (Atm "p") ⊢ FS (Atm "p")).

  Definition frefl (A : formula) : rule := ([],
                                            FS A ⊢ FS A).

  Definition refl : rule := ([],
                              FS (FV "A") ⊢ FS (FV "A")).

End CommonRules.
