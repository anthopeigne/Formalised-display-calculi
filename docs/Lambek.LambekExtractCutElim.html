Definition ex_rule : rule := ([], £((%"p" ∕ %"q") ⊗ %"q") ⊢ £((%"q" ∕ %"p") \ %"q")).

#[export] Instance derr_ex_rule : DerivRule Lambek_DC ex_rule.
Proof.
  Import LambekDeriv.
  set (p := %"p"). set (q := %"q").
  confirm_derr (Der (£((p ∕ q) ⊗ q) ⊢ £((q ∕ p) \ q)) CUT
                 [Der (£((p ∕ q) ⊗ q) ⊢ £p) Fusl
                 [Der (£(p ∕ q) ;; £q ⊢ £p) Rlesmr
                 [Der (£(p ∕ q) ⊢ £p ⟨ £q) Ovel
                   [Der (£p ⊢ £p) atrefl [];
                    Der (£q ⊢ £q) atrefl []]]];
                  Der (£p ⊢ £((q ∕ p) \ q)) Undr
                   [Der (£p ⊢ £(q ∕ p) ⟩ £q) Rsmgel
                   [Der (£(q ∕ p) ;; £p ⊢ £q) Rlesmr
                   [Der (£(q ∕ p) ⊢ £q ⟨ £p) Ovel
                     [Der (£q ⊢ £q) atrefl [];
                      Der (£p ⊢ £p) atrefl []]]]]]).
Defined.
